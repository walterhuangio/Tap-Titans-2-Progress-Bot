	#loopCount = #loopCount + 1

		#seconds = (#time/1000)%60
		#secondsTotal = #time/1000
		#minutes = (#time/1000)/60
	endif

		touchDown 0 100 625

		touchDown 0 350 630
		sleep 250
		touchUp 0
		sleep 250
	endif

		goto :pressSkills
	endif

		goto :checkClanQuest
	elseif #loopCount == 2
		goto :init
	elseif #minutes >= #prestigeAfterXMinutes and #enableAutoprestige == 1
		goto :checkPrestige
	elseif #timeDiff >= #runActionsEveryXMinutes
		goto :runActions

		goto :levelHeroes
